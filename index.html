Explanation: use while loop that is execute y times

#include // stdio.h

int main()

{
int x,y,j,t;
printf("enter the value of x:");
scanf("%d",&x);
printf("enter the value of y:");
scanf("%d",&y);
j=1;
t=y;
while(t!=0)
{
j=j*x;
t--;
}
printf("%d^%d:%d",x,y,j);
return 0;
}
